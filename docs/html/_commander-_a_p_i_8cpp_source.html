81bool Commander::executeCommand( const char *cmd, void* parent ){
82
83 // The beginning of the argument list will be stored in this pointer
84 char *arg;
85
86 // This variable tracks the command name length
87 uint32_t cmd_name_cntr;
88
89 // If this flag is set, than the description message will be printed,
90 // and the commands function won't be called.
91 uint8_t show_description = 0;
92
93 // Pointer to the selected command data.
94 systemCommand_t* commandData_ptr;
95
96 int pipePos;
97
98 bool executionStat;
99
100 int i;
101
102 // Copy the command data to the internal buffer.
103 // It is necessary because we have to modify the content
104 // of it. If it is points to a const char array we will
105 // get a bus-fault error without a buffer.
106 strncpy( tempBuff, cmd, COMMANDER_MAX_COMMAND_SIZE );
107
108 // According to the finding in issue #19 this has to be added to make things more safe.
109 tempBuff[ sizeof( tempBuff ) - 1 ] = '\0';
110
111 // Try to find a pipe character in the input string.
112 pipePos = hasChar( tempBuff, '|', true );
113
114 if( pipePos >= 0 ){
115
116 // Check if piping is requested, but disabled.
117 if( pipeArgBuffer == NULL ){
118
119 if( response != NULL ){
120 response -> print( __CONST_TXT__( "The command contains piping but the pipe module is disabled!" ) );
121 }
122 return false;
123 }
124
125 // Terminate where pip is found.
126 tempBuff[ pipePos ] = '\0';
127
128 }
129
130 // tempBuff is the address of the first character of the incoming command.
131 // If we give arg variable the value stored in tempBuff means arg will point to
132 // the first character of the command as well.
133 arg = tempBuff;
134
135 // Reset the name counter before we start counting
136 cmd_name_cntr = 0;
137
138 // Find the first space, question mark or a string-end character.
139 // At this time count how long is the command name( in characters )
140 while( ( *arg != '\0' ) && ( *arg != ' ' ) && ( *arg != '?' ) ){
141
142 cmd_name_cntr++;
143 arg++;
144
145 }
146
147 // If a space character found we have to terminate the string there.
148 // It is important because strcmp function will search for string terminator
149 // character, and this way we can separate the command name from its arguments.
150 if( *arg == ' ' ){
151
152 *arg = '\0';
153 arg++;
154
155 }
156
157 // The process is the same as above. The only difference is that this time
158 // we have to set the show_description flag.
159 else if( *arg == '?' ){
160
161 *arg = '\0';
162 arg++;
163 show_description = 1;
164
165 }
166
167 // Try to find the command datata.
168 commandData_ptr = regularCommands[ tempBuff ];
169
170 // If it is not a NULL pointer, that means we have a match.
171 if( commandData_ptr ){
172
173 // Because we have found the command in the API tree we have to choose
174 // between description printing and executing.
175 // If show_description flag is set, than we have to print the description.
176 if( show_description ){
177
178 if( memoryType == MEMORY_REGULAR ){
179
180 if( response != NULL ){
181
182 // Print the description text to the output channel.
183 response -> print( commandData_ptr -> name );
184 response -> print( ':' );
185 response -> print( ' ' );
186 response -> println( commandData_ptr -> data.desc );
187
188 }
189
190 }
191
192 #ifdef __AVR__
193
194 else if( memoryType == MEMORY_PROGMEM ){
195
196 if( response != NULL ){
197
198 // Print the description text to the output channel.
199 response -> print( commandData_ptr -> name_P );
200 response -> print( ':' );
201 response -> print( ' ' );
202 response -> println( commandData_ptr -> desc_P );
203
204 }
205
206 }
207
208 #endif
209
210
211 }
212
213 // If show_description flag is not set, than we have to execute the commands function.
214 else{
215
216 // Check if piping is enabled.
217 // If so, check if we have any data in the pipe channel.
218 if( ( pipeArgBuffer != NULL ) && ( pipeChannel -> available() > 0 ) ){
219
220 i = 0;
221
222 // Copy all the data from the pipe channel to the pipe buffer.
223 while( pipeChannel -> available() ){
224
225 // Check and handle overflow.
226 if( i < COMMANDER_MAX_COMMAND_SIZE ){
227 pipeArgBuffer[ i ] = pipeChannel -> read();
228 }
229
230 else{
231 pipeChannel -> read();
232 }
233
234 i++;
235
236 }
237
238 // If the data fits in the buffer, terminate after the last valid data.
239 if( i < COMMANDER_MAX_COMMAND_SIZE ){
240 pipeArgBuffer[ i ] = '\0';
241 }
242
243 // Just in case terminate at the end of the pipe buffer.
244 pipeArgBuffer[ COMMANDER_MAX_COMMAND_SIZE - 1 ] = '\0';
245
246 // Use the pipe buffer as argument.
247 arg = pipeArgBuffer;
248
249 }
250
251 // Check if piping is enabled.
252 // Check if we have at least one pipe left.
253 if( ( pipeArgBuffer != NULL ) && ( pipePos > 0 ) ){
254
255 // Execute commands function and redirect the output to the pipe channel.
256 executionStat = ( commandData_ptr -> data.func )( arg, pipeChannel, parent );
257
258 // If the actual command execution was successful, we have to increment the pipeCounter
259 if( executionStat ){
260 pipeCounter++;
261 }
262
263 }
264
265 // If piping is disabled or we ran out of pipes in the command string,
266 // we have to execute the command in the old fashion way.
267 else{
268
269 // Execute command function.
270 executionStat = ( commandData_ptr -> data.func )( arg, response, parent );
271
272 }
273
274 // Check if we had a problem during execution.
275 // If there is a problem, we have to return with error code.
276 if( executionStat == false ){
277
278 /*
279 response -> print( "pipeCounter: " );
280 response -> println( pipeCounter );
281 */
282
283 // Check if the piping went wrong.
284 int brokenPipePos = hasChar( originalCommandData, '|', pipeCounter, true );
285
286 if( brokenPipePos >= 0 ){
287
288 if( response != NULL ){
289
290 // In case of broken pipe, inform the user about the problematic section.
291 if( formatting ){
292 response -> print( __CONST_TXT__( "\033[1;35m" ) );
293 }
294
295 response -> println( __CONST_TXT__( "\r\nBroken pipe!" ) );
296
297 if( formatting ){
298 response -> print( __CONST_TXT__( "\033[0;37m" ) );
299 }
300
301 response -> println( originalCommandData );
302
303 for( i = 0; i < brokenPipePos; i++ ){
304 response -> print( ' ' );
305 }
306
307 if( formatting ){
308 response -> print( __CONST_TXT__( "\033[1;31m" ) );
309 }
310
311 response -> println( "\u25B2" );
312
313 for( i = 0; i < brokenPipePos; i++ ){
314 response -> print( ' ' );
315 }
316
317 response -> print( "\u2514 The pipe broke here" );
318
319 if( formatting ){
320 response -> print( __CONST_TXT__( "\033[0;37m" ) );
321 }
322
323 }
324
325 }
326
327 return false;
328 }
329
330 // Check if piping is enabled.
331 // Check if we have at least one pipe left.
332 if( ( pipeArgBuffer != NULL ) && pipePos > 0 ){
333
334 // To remove whitespace from the new command begin.
335 while( tempBuff[ pipePos + 1 ] == ' ' ){
336 pipePos++;
337 }
338
339 // Execute the next command in the pipe.
340 if( executeCommand( &tempBuff[ pipePos + 1 ] ) ){
341 // If the command execution fails, we have to return to break recursion.
342 return false;
343 }
344
345 }
346
347 }
348
349 }
350
351 // If it is not an added function, we have to check for internal functions.
352 // 'help' is an internal function that prints the available commands in order.
353 else if( strcmp( tempBuff, (const char*)"help" ) == 0 ){
354
355 // We have to check for single or described help function.
356 if( strcmp( arg, (const char*)"-d" ) == 0 ){
357
358 printHelp( response, true, formatting );
359
360 }
361
362 else{
363
364 printHelp( response, false, formatting );
365
366 }
367
368 }
369
370 else{
371
372 if( response != NULL ){
373
374 // If we went through the whole tree and we did not found the command in it,
375 // we have to notice the user abut the problem. Maybe a Type-O
376 response -> print( __CONST_TXT__( "Command \'" ) );
377 response -> print( tempBuff );
378 response -> print( __CONST_TXT__( "\' not found!" ) );
379
380 }
381
382 }
383 return true;
384}

386bool Commander::execute( const char *cmd ){
387
388 // Default execute handler, so the default response will be chosen.
389 response = NULL;
390
391 // Reset the pipe position tracker.
392 pipeCounter = 0;
393
394 // Save the address of the original message.
395 originalCommandData = cmd;
396
397 // If piping is enabled, we have to flush the pipe channel before command execution.
398 if( pipeArgBuffer != NULL ){
399 while( pipeChannel -> available() ){
400 pipeChannel -> read();
401 }
402 }
403
404 // Execute the command.
405 return( executeCommand( cmd ) );
406

441
442 uint32_t i;
443
444 if( style ){
445 out -> println( __CONST_TXT__( "\033[1;31m----\033[1;32m Available commands \033[1;31m----\033[0;37m\r\n" ) );

448 out -> println( __CONST_TXT__( "---- Available commands ----\r\n" ) );

452

493int Commander::hasChar( const char* str, char c, bool ignoreString ){
494
495 int cntr = 0;
496
497 while( str[ cntr ] ){
498
499 // Check if we have to ignore characters inside a string.
500 // If so, we have to check if we are in a string.
501 if( ignoreString && inString( str, cntr ) ){
502 // In this case we can skip to the next character.
503 cntr++;
504 continue;
505 }
506
507 // Check if the current character is the searched one.
508 if( str[ cntr ] == c ){
509 // If it is, return with its position( index ).
510 return cntr;
511
512 }
513
514 // Increment the counter.
515 cntr++;
516
517 }
518
519 // If we are here, that means, we did not found the character in the string :(
520 // Return with an invalid address.
521 return -1;
522
523}

525int Commander::hasChar( const char* str, char c, int number, bool ignoreString ){
526
527 // Generic counter.
528 int i;
529
530 // It will store the current result.
531 int result;
532
533 // It will store te total position.
534 int cntr = 0;
535
536 // Try to find the number'th occurrence in the input string.
537 for( i = 0; i < number; i++ ){
538
539 result = hasChar( &str[ cntr ], c, ignoreString );
540
541 if( result < 0 ){
542 return -1;
543 }
544
545 cntr += ( result + 1 );
546
547 }
548
549 return cntr - 1;
550
551}

681bool Commander::inString( const char* source, int index ){
682
683 // Generic counter.
684 int i;
685
686 // It will store the return value.
687 bool ret = false;
688
689 // Check if the source is not set correctly.
690 if( source == NULL ){
691 return false;
692 }
693
694 // Check if the index is not set correctly.
695 if( index < 0 ){
696 return false;
697 }
698
699 if( index >= (int)strlen( source ) ){
700 return false;
701 }
702
703 // Go through every character from the first to the indexed one.
704 for( i = 0; i < index; i++ ){
705
706 // Every time we find a string start/end character, which is the " character,
707 // we negate the return value.
708 if( source[ i ] == '\"' ){
709 ret = !ret;
710 }
711
712 }
713
714 // Return with the result.
715 return ret;
716
717}

719void Commander::update( char* buffer, int bufferSize, Stream* channel_p ){
720
721 // It will hold the nex incoming character from the channel.
722 char c;
723
724 if( bufferSize <=0 ){
725 return;
726 }
727
728 if( channel_p == NULL ){
729 return;
730 }
731
732 // Check if there is any data incoming.
733 while( channel_p -> available() ){
734
735 // Read the next incoming character.
736 c = channel_p -> read();
737
738 // Every command from Serial is terminated with a new-line
739 // character. If a new-line character arrives, we have to
740 // terminate the string in the commandFromSerial buffer,
741 // and execute it. After execution, we have to reset the
742 // commandIndex counter to zero.
743 if( c == '\r' ){
744 buffer[ updateBufferCounter ] = '\0';
745 channel_p -> println();
746 execute( buffer, channel_p );
747 updateBufferCounter = 0;
748 channel_p -> print( "\r\n$: " );
749 }
750
751 // If we have a carriage-return character we simply
752 // ignore it.
753 else if( c == '\n' ){
754 continue;
755 }
756
757 // Handle backspace events.
758 else if( ( c == '\b' ) || ( c == 127 ) ){
759 if( updateBufferCounter > 0 ){
760 updateBufferCounter--;
761 channel_p -> print( "\b \b" );
762 }
763 }
764
765 // Every other case we just put the data to the next
766 // free space in the commandFromSerial buffer, increment
767 // the commandIndex, and check if it wants to overflow.
768 else{
769 buffer[ updateBufferCounter ] = c;
770 updateBufferCounter++;
771 if( updateBufferCounter >= bufferSize ){
772 updateBufferCounter = bufferSize - 1;
773 }
774 else{
775 channel_p -> print( c );
776 }
777 }
778
779 }

783int Commander::floatToString( float number, char* buffer, int bufferSize ){
784
785 // Used to generate string from float.
786 bool negative = false;
787
788 // This variable is used to round the float value correctly.
789 float roundVal;
790
791 // This will hold the decimal part of the float number.
792 int decimalPart;
793
794 // This will hold the fraction part of the float number.
795 int fractionPart;
796
797 // Return status for snprintf;
798 int status;
799
800 if( bufferSize < 2 ){
801 return -1;
802 }
803
804 // Check if the number is negative.
805 // If it is, set the negative flag,
806 // and make it positive.
807 if( number < 0 ){
808 negative = true;
809 number = -number;
810 }
811
812 // Round the result correctly.
813 // For example 9.999 with precision of 2 will be printed as 10.00.
814 roundVal = 0.5f;
815 roundVal /= 100.0;
816 number += roundVal;
817
818 // Split the number to decimal and fraction parts
819 decimalPart = (int)number;
820 fractionPart = (int)( ( number - (float)decimalPart ) * 100.0f );
821
822 // If the number is negative, put a '-' character to the first element
823 // of the buffer and increment its position. Also decrement the buffer
824 // size, because we used the first position.
825 if( negative ){
826 buffer[ 0 ] = '-';
827 buffer++;
828 bufferSize--;
829 }
830 else{
831 status = snprintf( buffer, bufferSize, "%d.%0*d", decimalPart, 2, fractionPart );
832 }
833
834 // If we had to print a '-' character because a negative number,
835 // we had to increment the status, if the status is not 0 or negative.
836 if( ( status > 0 ) && negative ){
837 status++;
838 }
839
840 return status;
841
842}
