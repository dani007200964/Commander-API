87bool Commander::executeCommand( const char *cmd ){
88
89 // The beginning of the argument list will be stored in this pointer
90 char *arg;
91
92 // This variable tracks the command name length
93 uint32_t cmd_name_cntr;
94
95 // If this flag is set, than the description message will be printed,
96 // and the commands function won't be called.
97 uint8_t show_description = 0;
98
99 // Pointer to the selected command data.
100 systemCommand_t* commandData_ptr;
101
102 int pipePos;
103
104 bool executionStat;
105
106 int i;
107
108 // Copy the command data to the internal buffer.
109 // It is necessary because we have to modify the content
110 // of it. If it is points to a const char array we will
111 // get a bus-fault error without a buffer.
112 strncpy( tempBuff, cmd, COMMANDER_MAX_COMMAND_SIZE );
113
114 // According to the finding in issue #19 this has to be added to make things more safe.
115 tempBuff[ sizeof( tempBuff ) - 1 ] = '\0';
116
117 // Try to find a pipe character in the input string.
118 pipePos = hasChar( tempBuff, '|', true );
119
120 if( pipePos >= 0 ){
121
122 // Check if piping is requested, but disabled.
123 if( pipeArgBuffer == NULL ){
124
125 //if( caller -> channel != NULL ){
126 caller -> println( __CONST_TXT__( "The command contains piping but the pipe module is disabled!" ) );
127 //}
128 return false;
129 }
130
131 // Terminate where pipe is found.
132 tempBuff[ pipePos ] = '\0';
133
134 }
135
136 // tempBuff is the address of the first character of the incoming command.
137 // If we give arg variable the value stored in tempBuff means arg will point to
138 // the first character of the command as well.
139 arg = tempBuff;
140
141 // Reset the name counter before we start counting
142 cmd_name_cntr = 0;
143
144 // Find the first space, question mark or a string-end character.
145 // At this time count how long is the command name( in characters )
146 while( ( *arg != '\0' ) && ( *arg != ' ' ) && ( *arg != '?' ) ){
147
148 cmd_name_cntr++;
149 arg++;
150
151 }
152
153 // If a space character found we have to terminate the string there.
154 // It is important because strcmp function will search for string terminator
155 // character, and this way we can separate the command name from its arguments.
156 if( *arg == ' ' ){
157
158 *arg = '\0';
159 arg++;
160
161 }
162
163 // The process is the same as above. The only difference is that this time
164 // we have to set the show_description flag.
165 else if( *arg == '?' ){
166
167 *arg = '\0';
168 arg++;
169 show_description = 1;
170
171 }
172
173 // Try to find the command datata.
174 commandData_ptr = regularCommands[ tempBuff ];
175
176 // If it is not a NULL pointer, that means we have a match.
177 if( commandData_ptr ){
178
179 // Because we have found the command in the API tree we have to choose
180 // between description printing and executing.
181 // If show_description flag is set, than we have to print the description.
182 if( show_description ){
183
184 #ifdef __AVR__
185 caller -> print( commandData_ptr -> name );
186 caller -> print( ':' );
187 caller -> print( ' ' );
188 if( commandData_ptr -> data.desc ){
189 caller -> print( commandData_ptr -> data.desc );
190 }
191 else if( commandData_ptr -> data.desc_P ){
192 caller -> print( commandData_ptr -> data.desc_P );
193 }
194 caller -> println();
195 #else
196 caller -> print( commandData_ptr -> name );
197 caller -> print( ':' );
198 caller -> print( ' ' );
199 caller -> println( commandData_ptr -> data.desc );
200 #endif
201
202 }
203
204 // If show_description flag is not set, than we have to execute the commands function.
205 else{
206
207 // Check if piping is enabled.
208 // If so, check if we have any data in the pipe channel.
209 if( ( pipeArgBuffer != NULL ) && ( pipeChannel -> available() > 0 ) ){
210
211 i = 0;
212
213 // Copy all the data from the pipe channel to the pipe buffer.
214 while( pipeChannel -> available() ){
215
216 // Check and handle overflow.
217 if( i < COMMANDER_MAX_COMMAND_SIZE ){
218 pipeArgBuffer[ i ] = pipeChannel -> read();
219 }
220
221 else{
222 pipeChannel -> read();
223 }
224
225 i++;
226
227 }
228
229 // If the data fits in the buffer, terminate after the last valid data.
230 if( i < COMMANDER_MAX_COMMAND_SIZE ){
231 pipeArgBuffer[ i ] = '\0';
232 }
233
234 // Just in case terminate at the end of the pipe buffer.
235 pipeArgBuffer[ COMMANDER_MAX_COMMAND_SIZE - 1 ] = '\0';
236
237 // Use the pipe buffer as argument.
238 arg = pipeArgBuffer;
239
240 }
241
242 // Check if piping is enabled.
243 // Check if we have at least one pipe left.
244 if( ( pipeArgBuffer != NULL ) && ( pipePos > 0 ) ){
245
246 // Execute commands function and redirect the output to the pipe channel.
247 Stream* tmp = caller -> getChannel();
248 caller -> setChannel( pipeChannel );
249 executionStat = ( commandData_ptr -> data.func )( arg, caller );
250 caller -> setChannel( tmp );
251
252 // If the actual command execution was successful, we have to increment the pipeCounter
253 if( executionStat ){
254 pipeCounter++;
255 }
256
257 }
258
259 // If piping is disabled or we ran out of pipes in the command string,
260 // we have to execute the command in the old fashion way.
261 else{
262
263 // Execute command function.
264 executionStat = ( commandData_ptr -> data.func )( arg, caller );
265
266 }
267
268 // Check if we had a problem during execution.
269 // If there is a problem, we have to return with error code.
270 if( executionStat == false ){
271
272 /*
273 response -> print( "pipeCounter: " );
274 response -> println( pipeCounter );
275 */
276
277 /*
278
279 // Check if the piping went wrong.
280 brokenPipePos = hasChar( originalCommandData, '|', pipeCounter, true );
281
282 if( brokenPipePos >= 0 ){
283
284 if( response != NULL ){
285
286 // In case of broken pipe, inform the user about the problematic section.
287 if( formatting ){
288 response -> print( __CONST_TXT__( "\033[1;35m" ) );
289 }
290
291 response -> println( __CONST_TXT__( "\r\nBroken pipe!" ) );
292
293 if( formatting ){
294 response -> print( __CONST_TXT__( "\033[0;37m" ) );
295 }
296
297 response -> println( originalCommandData );
298
299 for( i = 0; i < brokenPipePos; i++ ){
300 response -> print( ' ' );
301 }
302
303 if( formatting ){
304 response -> print( __CONST_TXT__( "\033[1;31m" ) );
305 }
306
307 response -> println( "\u25B2" );
308
309 for( i = 0; i < brokenPipePos; i++ ){
310 response -> print( ' ' );
311 }
312
313 response -> print( "\u2514 The pipe broke here" );
314
315 if( formatting ){
316 response -> print( __CONST_TXT__( "\033[0;37m" ) );
317 }
318
319 }
320
321 }
322
323 */
324
325 if( pipePos > 0 ){
326 tempBuff[ pipePos ] = '|';
327 }
328
329 printBrokenPipe();
330
331 return false;
332 }
333
334 // Check if piping is enabled.
335 // Check if we have at least one pipe left.
336 if( ( pipeArgBuffer != NULL ) && pipePos > 0 ){
337
338 // To remove whitespace from the new command begin.
339 while( tempBuff[ pipePos + 1 ] == ' ' ){
340 pipePos++;
341 }
342
343 // Execute the next command in the pipe.
344 if( !executeCommand( &tempBuff[ pipePos + 1 ] ) ){
345 // If the command execution fails, we have to return to break recursion.
346 return false;
347 }
348
349 }
350
351 }
352
353 }
354
355 // If it is not an added function, we have to check for internal functions.
356 // 'help' is an internal function that prints the available commands in order.
357 else if( strcmp( tempBuff, (const char*)"help" ) == 0 ){
358
359 // We have to check for single or described help function.
360 if( strcmp( arg, (const char*)"-d" ) == 0 ){
361
362 printHelp( caller, true, formatting );
363
364 }
365
366 else{
367
368 printHelp( caller, false, formatting );
369
370 }
371
372 }
373
374 else{
375
376 //if( caller -> channel != NULL ){
377 if( pipePos > 0 ){
378 tempBuff[ pipePos ] = '|';
379 printBrokenPipe();
380 }
381 else if( pipeCounter > 0 ){
382 printBrokenPipe();
383
384 }
385
386 // If we went through the whole tree and we did not found the command in it,
387 // we have to notice the user abut the problem. Maybe a Type-O
388 caller -> print( __CONST_TXT__( "Command \'" ) );
389 caller -> print( tempBuff );
390 caller -> println( __CONST_TXT__( "\' not found!" ) );
391
392 //}
393 return false;
394
395 }
396 return true;

399bool Commander::execute( const char *cmd, Stream* channel_p, CommandCaller* caller_p ){
400
401 if( channel_p == NULL ){
402 caller = &defaultCommandCaller;
403 caller -> clearChannel();
404 }
405
406 else{
407 if( caller_p == NULL ){
408 caller = &defaultCommandCaller;
409 caller -> setChannel( channel_p );
410 }
411 else{
412 caller = caller_p;
413 }
414 }
415
416 // Reset the pipe position tracker.
417 pipeCounter = 0;
418
419 // Save the address of the original message.
420 originalCommandData = cmd;
421
422 // If piping is enabled, we have to flush the pipe channel before command execution.
423 if( pipeArgBuffer != NULL ){
424 while( pipeChannel -> available() ){
425 pipeChannel -> read();
426 }
427 }
428
429 // Execute the command.
430 return( executeCommand( cmd ) );
431

495int Commander::hasChar( const char* str, char c, bool ignoreString ){
496
497 int cntr = 0;
498
499 while( str[ cntr ] ){
500
501 // Check if we have to ignore characters inside a string.
502 // If so, we have to check if we are in a string.
503 if( ignoreString && inString( str, cntr ) ){
504 // In this case we can skip to the next character.
505 cntr++;
506 continue;
507 }
508
509 // Check if the current character is the searched one.
510 if( str[ cntr ] == c ){
511 // If it is, return with its position( index ).
512 return cntr;
513
514 }
515
516 // Increment the counter.
517 cntr++;
518
519 }
520
521 // If we are here, that means, we did not found the character in the string :(
522 // Return with an invalid address.
523 return -1;
524
525}

527int Commander::hasChar( const char* str, char c, int number, bool ignoreString ){
528
529 // Generic counter.
530 int i;
531
532 // It will store the current result.
533 int result;
534
535 // It will store te total position.
536 int cntr = 0;
537
538 // Try to find the number'th occurrence in the input string.
539 for( i = 0; i < number; i++ ){
540
541 result = hasChar( &str[ cntr ], c, ignoreString );
542
543 if( result < 0 ){
544 return -1;
545 }
546
547 cntr += ( result + 1 );
548
549 }
550
551 return cntr - 1;

666bool Commander::inString( const char* source, int index ){
667
668 // Generic counter.
669 int i;
670
671 // It will store the return value.
672 bool ret = false;
673
674 // Check if the source is not set correctly.
675 if( source == NULL ){
676 return false;
677 }
678
679 // Check if the index is not set correctly.
680 if( index < 0 ){
681 return false;
682 }
683
684 if( index >= (int)strlen( source ) ){
685 return false;
686 }
687
688 // Go through every character from the first to the indexed one.
689 for( i = 0; i < index; i++ ){
690
691 // Every time we find a string start/end character, which is the " character,
692 // we negate the return value.
693 if( source[ i ] == '\"' ){
694 ret = !ret;
695 }
696
697 }
698
699 // Return with the result.
700 return ret;
701
702}

704void Commander::update( char* buffer, int bufferSize, Stream* channel_p ){
705
706 // It will hold the nex incoming character from the channel.
707 char c;
708
709 if( bufferSize <=0 ){
710 return;
711 }
712
713 if( channel_p == NULL ){
714 return;
715 }
716
717 // Check if there is any data incoming.
718 while( channel_p -> available() ){
719
720 // Read the next incoming character.
721 c = channel_p -> read();
722
723 // Every command from Serial is terminated with a new-line
724 // character. If a new-line character arrives, we have to
725 // terminate the string in the commandFromSerial buffer,
726 // and execute it. After execution, we have to reset the
727 // commandIndex counter to zero.
728 if( c == '\r' ){
729 buffer[ updateBufferCounter ] = '\0';
730 channel_p -> println();
731 execute( buffer, channel_p );
732 updateBufferCounter = 0;
733 channel_p -> print( "\r\n$: " );
734 }
735
736 // If we have a carriage-return character we simply
737 // ignore it.
738 else if( c == '\n' ){
739 continue;
740 }
741
742 // Handle backspace events.
743 else if( ( c == '\b' ) || ( c == 127 ) ){
744 if( updateBufferCounter > 0 ){
745 updateBufferCounter--;
746 channel_p -> print( "\b \b" );
747 }
748 }
749
750 // Every other case we just put the data to the next
751 // free space in the commandFromSerial buffer, increment
752 // the commandIndex, and check if it wants to overflow.
753 else{
754 buffer[ updateBufferCounter ] = c;
755 updateBufferCounter++;
756 if( updateBufferCounter >= bufferSize ){
757 updateBufferCounter = bufferSize - 1;
758 }
759 else{
760 channel_p -> print( c );
761 }
762 }
763
764 }
765
766}

768int Commander::floatToString( float number, char* buffer, int bufferSize ){
769
770 // Used to generate string from float.
771 bool negative = false;
772
773 // This variable is used to round the float value correctly.
774 float roundVal;
775
776 // This will hold the decimal part of the float number.
777 int decimalPart;
778
779 // This will hold the fraction part of the float number.
780 int fractionPart;
781
782 // Return status for snprintf;
783 int status;
784
785 if( bufferSize < 2 ){
786 return -1;
787 }
788
789 // Check if the number is negative.
790 // If it is, set the negative flag,
791 // and make it positive.
792 if( number < 0 ){
793 negative = true;
794 number = -number;
795 }
796
797 // Round the result correctly.
798 // For example 9.999 with precision of 2 will be printed as 10.00.
799 roundVal = 0.5f;
800 roundVal /= 100.0;
801 number += roundVal;
802
803 // Split the number to decimal and fraction parts
804 decimalPart = (int)number;
805 fractionPart = (int)( ( number - (float)decimalPart ) * 100.0f );
806
807 // If the number is negative, put a '-' character to the first element
808 // of the buffer and increment its position. Also decrement the buffer
809 // size, because we used the first position.
810 if( negative ){
811 buffer[ 0 ] = '-';
812 buffer++;
813 bufferSize--;
814 }
815 else{
816 status = snprintf( buffer, bufferSize, "%d.%02d", decimalPart, fractionPart );
817 }
818
819 // If we had to print a '-' character because a negative number,
820 // we had to increment the status, if the status is not 0 or negative.
821 if( ( status > 0 ) && negative ){
822 status++;
823 }
824
825 return status;
826
827}

928int Commander::generateHint( const char *fraction, char *buffer_p, int buffer_size_p ){
929
930 int char_index;
931 int prev_char_index;
932 int i;
933 int number_of_hints;
934
935 // Pointer to the selected command data.
936 systemCommand_t* commandData_ptr;
937
938 lastHint = NULL;
939 lastCommandHint = NULL;
940 lastCommandHintOffset = 0;
941
942 // Check if we have an empty string.
943 if( fraction[ 0 ] == '\0' ){
944 return 0;
945 }
946
947 // Copy the command data to the internal buffer.
948 // It is necessary because we have to modify the content
949 // of it. If it is points to a const char array we will
950 // get a bus-fault error without a buffer.
951 strncpy( tempBuff, fraction, COMMANDER_MAX_COMMAND_SIZE );
952
953 // According to the finding in issue #19 this has to be added to make things more safe.
954 tempBuff[ sizeof( tempBuff ) - 1 ] = '\0';
955
956 // Try to find the first space character.
957 // If we have at least one space characters, that
958 // means, we have a command typed in already.
959 char_index = hasChar( tempBuff, ' ' );
960 if( char_index >= 0 ){
961 tempBuff[ char_index ] = '\0';
962 if( !commandExists( tempBuff, &commandData_ptr ) ){
963 return 0;
964 }
965
966 // If we are here, that means, we found a valid command.
967 // Let's try to extract some help to the command.
968 // Check if help data is added to the object.
969 lastHint = commandData_ptr -> data.help;
970 if( lastHint == NULL ){
971 return 0;
972 }
973
974 // If we have some help, we need to find the second last
975 // space character.
976
977 // Change back the \0 character to the original state.
978 tempBuff[ char_index ] = ' ';
979
980 i = char_index;
981 while( tempBuff[ i ] ){
982 if( tempBuff[ i ] == ' ' ){
983 prev_char_index = char_index;
984 char_index = i;
985 }
986 i++;
987 }
988
989 // Try to generate a hint from the second last space position. For example 'cat -d '.
990 number_of_hints = lastHint -> generateHint( &tempBuff[ prev_char_index ], buffer_p, buffer_size_p );
991
992 // Check if we gat any help. If not, try with the last space character. For example ' cat -d "txt" '.
993 if( number_of_hints <= 0 ){
994 number_of_hints = lastHint -> generateHint( &tempBuff[ char_index ], buffer_p, buffer_size_p );
995 }
996
997 // Check if we gat any help. If not, try to list anything at all.
998 if( number_of_hints <= 0 ){
999 number_of_hints = lastHint -> generateHint( " ", buffer_p, buffer_size_p );
1000 }
1001
1002 return number_of_hints;
1003
1004 }
1005
1006 // If we are here, that means, we has a command which is not
1007 // typed fully.
1008
1009 lastCommandHintOffset = strlen( tempBuff );
1010 number_of_hints = regularCommands.completeFragment( tempBuff, buffer_p, buffer_size_p );
1011 if( number_of_hints > 0 ){
1012 lastCommandHint = buffer_p;
1013 }
1014 return number_of_hints;
