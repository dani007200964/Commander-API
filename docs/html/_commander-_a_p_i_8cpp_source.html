50void Commander::init(){
51
52 // Generic conter variables.
53 uint32_t i;
54 uint32_t j;
55
56 // Temporary variable, used to flip elements.
57 API_t temp;
58
59 commander_strcmp = &Commander::commander_strcmp_regular;
60 commander_strcmp_tree_ram = &Commander::commander_strcmp_tree_ram_regular;
61
62 #ifdef __AVR__
63
64 if( API_tree[ 0 ].name == NULL ){
65
66 memoryType = MEMORY_PROGMEM;
67 commander_strcmp = &Commander::commander_strcmp_progmem;
68 commander_strcmp_tree_ram = &Commander::commander_strcmp_tree_ram_progmem;
69
70 }
71
72 #endif
73
74 dbgResponse -> println( __CONST_TXT__( "Commander init start" ) );
75
76 // Make the tree ordered by alphabet.
77 dbgResponse -> print( __CONST_TXT__( " Creating alphabetical order... " ) );
78
79 // Bubble sort. We need to sort the commands into an alphabetical order.
80 for( i = 0; i < API_tree_size; i++ ){
81
82 for( j = i + 1; j < API_tree_size; j++ ){
83
84 if( ( this ->* commander_strcmp )( &API_tree[ i ], &API_tree[ j ] ) > 0 ){
85
86 temp = API_tree[ i ];
87 API_tree[ i ] = API_tree[ j ];
88 API_tree[ j ] = temp;
89
90 }
91
92 }
93
94 }
95
96 // Fill the place variable in the tree with
97 // correct alphabetical place.
98 for( i = 0; i < API_tree_size; i++ ){
99
100 API_tree[ i ].place = i;
101
102 }
103
104 dbgResponse -> println( __CONST_TXT__( "[ OK ]" ) );
105
106 // Optimize the tree to make it balanced.
107 // It is necessary to speed up the command
108 // search phase.
109 dbgResponse -> print( __CONST_TXT__( " Create balanced binary structure... " ) );
110 optimize_api_tree();
111
112 dbgResponse -> println( __CONST_TXT__( "[ OK ]" ) );
113 dbgResponse -> println( __CONST_TXT__( "Commander init finished!" ) );
114
115}

145void Commander::swap_api_elements( uint16_t index, uint16_t place ){
146
147 // Buffer that will temporary hold an element.
148 // This is required for a swap.
149 API_t buffer;
150
151 // This variable will store the address of the element defined by the second argument( place ).
152 uint16_t current_index;
153
154 // Find the index in the array by place of the 'i'-th element
155 current_index = find_api_index_by_place( place );
156
157 // save the context of the 'i'-th element to the buffer
158 buffer = API_tree[index];
159
160 // write the 'current_index'-th element to the 'i'-th element
161 API_tree[index] = API_tree[current_index];
162
163 // write the buffer to the 'current_index'-th element
164 API_tree[current_index] = buffer;
165

168void Commander::optimize_api_tree(){
169
170 uint32_t i;
171
172 //API_t buffer;
173
174 // Stores the next elements address in the tree
175 API_t *next;
176
177 // Stores the previous elements address in the tree
178 API_t *prev;
179
180 // It will store string comparison result
181 int comp_res;
182
183 // recursive optimizer need to initialize elementCounter to 0
184 elementCounter = 0;
185
186 // recursively finds the order which is optimal for a balanced tree
187 recursive_optimizer( 0, API_tree_size - 1 );
188
189 // The order is good, but the connection between the branches broken,
190 // because we swapped the API_tree array elements.
191 // To fix this problem we have to add elements from index 1 and
192 // place them in the binary tree.
193 for( i = 1; i < API_tree_size; i++ ){
194
195 prev = &API_tree[ 0 ];
196
197 comp_res = ( this ->* commander_strcmp )( prev, &API_tree[ i ] );
198
199 (comp_res > 0) ? (next = (prev->left)) : ( next = (prev->right));
200
201 while( next != NULL ){
202
203 prev = next;
204 comp_res = ( this ->* commander_strcmp )( prev, &API_tree[ i ] );
205 (comp_res > 0) ? (next = (prev->left)) : ( next = (prev->right));
206
207 }
208
209 ( comp_res > 0 ) ? ( ( prev->left ) = &API_tree[ i ] ) : ( ( prev->right ) = &API_tree[ i ] );
210
211 }
212
213}

215void Commander::recursive_optimizer( int32_t start_index, int32_t stop_index ){
216
217 // The middle number between start and stop index
218 // will be stored in this variable.
219 int32_t mid;
220
221 // Detect the end of recursion.
222 if( start_index > stop_index ){
223
224 return;
225
226 }
227
228 // Find the middle of the interval
229 mid = ( start_index + stop_index ) / 2;
230
231 // Put the right element to it's place
232 swap_api_elements( elementCounter, mid );
233 elementCounter++;
234
235 // Do some recursion for the other intervals
236 recursive_optimizer( start_index, mid - 1 );
237 recursive_optimizer( mid + 1, stop_index );
238
239
240}

263bool Commander::executeCommand( const char *cmd, void* parent ){
264
265 // The beginning of the argument list will be stored in this pointer
266 char *arg;
267
268 // This variable tracks the command name length
269 uint32_t cmd_name_cntr;
270
271 // If this flag is set, than the description message will be printed,
272 // and the commands function won't be called.
273 uint8_t show_description = 0;
274
275 // Pointer to the selected command data.
276 API_t *commandData_ptr;
277
278 int pipePos;
279
280 bool executionStat;
281
282 int i;
283
284 // Copy the command data to the internal buffer.
285 // It is necessary because we have to modify the content
286 // of it. If it is points to a const char array we will
287 // get a bus-fault error without a buffer.
288 strncpy( tempBuff, cmd, COMMANDER_MAX_COMMAND_SIZE );
289
290 // According to the finding in issue #19 this has to be added to make things more safe.
291 tempBuff[ sizeof( tempBuff ) - 1 ] = '\0';
292
293 // Try to find a pipe character in the input string.
294 pipePos = hasChar( tempBuff, '|', true );
295
296 if( pipePos >= 0 ){
297
298 // Check if piping is requested, but disabled.
299 if( pipeArgBuffer == NULL ){
300 response -> print( __CONST_TXT__( "The command contains piping but the pipe module is disabled!" ) );
301 return false;
302 }
303
304 // Terminate where pip is found.
305 tempBuff[ pipePos ] = '\0';
306
307 }
308
309 // tempBuff is the address of the first character of the incoming command.
310 // If we give arg variable the value stored in tempBuff means arg will point to
311 // the first character of the command as well.
312 arg = tempBuff;
313
314 // Reset the name counter before we start counting
315 cmd_name_cntr = 0;
316
317 // Find the first space, question mark or a string-end character.
318 // At this time count how long is the command name( in characters )
319 while( ( *arg != '\0' ) && ( *arg != ' ' ) && ( *arg != '?' ) ){
320
321 cmd_name_cntr++;
322 arg++;
323
324 }
325
326 // If a space character found we have to terminate the string there.
327 // It is important because strcmp function will search for string terminator
328 // character, and this way we can separate the command name from its arguments.
329 if( *arg == ' ' ){
330
331 *arg = '\0';
332 arg++;
333
334 }
335
336 // The process is the same as above. The only difference is that this time
337 // we have to set the show_description flag.
338 else if( *arg == '?' ){
339
340 *arg = '\0';
341 arg++;
342 show_description = 1;
343
344 }
345
346 // Try to find the command datata.
347 commandData_ptr = (*this)[ tempBuff ];
348
349 // If it is not a NULL pointer, that means we have a match.
350 if( commandData_ptr ){
351
352 // Because we have found the command in the API tree we have to choose
353 // between description printing and executing.
354 // If show_description flag is set, than we have to print the description.
355 if( show_description ){
356
357 if( memoryType == MEMORY_REGULAR ){
358
359 // Print the description text to the output channel.
360 response -> print( commandData_ptr -> name );
361 response -> print( ':' );
362 response -> print( ' ' );
363 response -> println( commandData_ptr -> desc );
364
365 }
366
367 #ifdef __AVR__
368
369 else if( memoryType == MEMORY_PROGMEM ){
370
371 // Print the description text to the output channel.
372 response -> print( commandData_ptr -> name_P );
373 response -> print( ':' );
374 response -> print( ' ' );
375 response -> println( commandData_ptr -> desc_P );
376
377 }
378
379 #endif
380
381
382 }
383
384 // If show_description flag is not set, than we have to execute the commands function.
385 else{
386
387 // Check if piping is enabled.
388 // If so, check if we have any data in the pipe channel.
389 if( ( pipeArgBuffer != NULL ) && ( pipeChannel -> available() > 0 ) ){
390
391 i = 0;
392
393 // Copy all the data from the pipe channel to the pipe buffer.
394 while( pipeChannel -> available() ){
395
396 // Check and handle overflow.
397 if( i < COMMANDER_MAX_COMMAND_SIZE ){
398
399 pipeArgBuffer[ i ] = pipeChannel -> read();
400
401 }
402
403 else{
404
405 pipeChannel -> read();
406
407 }
408
409 i++;
410
411 }
412
413 // If the data fits in the buffer, terminate after the last valid data.
414 if( i < COMMANDER_MAX_COMMAND_SIZE ){
415
416 pipeArgBuffer[ i ] = '\0';
417
418 }
419
420 // Just in case terminate at the end of the pipe buffer.
421 pipeArgBuffer[ COMMANDER_MAX_COMMAND_SIZE - 1 ] = '\0';
422
423 // Use the pipe buffer as argument.
424 arg = pipeArgBuffer;
425
426 }
427
428 // Check if piping is enabled.
429 // Check if we have at least one pipe left.
430 if( ( pipeArgBuffer != NULL ) && ( pipePos > 0 ) ){
431
432 // Execute commands function and redirect the output to the pipe channel.
433 executionStat = (commandData_ptr -> func)( arg, pipeChannel, parent );
434
435 // If the actual command execution was successful, we have to increment the pipeCounter
436 if( executionStat ){
437 pipeCounter++;
438 }
439
440 }
441
442 // If piping is disabled or we ran out of pipes in the command string,
443 // we have to execute the command in the old fashion way.
444 else{
445
446 // Execute command function.
447 executionStat = (commandData_ptr -> func)( arg, response, parent );
448
449 }
450
451 // Check if we had a problem during execution.
452 // If there is a problem, we have to return with error code.
453 if( executionStat == false ){
454
455 /*
456 response -> print( "pipeCounter: " );
457 response -> println( pipeCounter );
458 */
459
460 // Check if the piping went wrong.
461 int brokenPipePos = hasChar( originalCommandData, '|', pipeCounter, true );
462
463 if( brokenPipePos >= 0 ){
464
465 // In case of broken pipe, inform the user about the problematic section.
466 if( formatting ){
467 response -> print( __CONST_TXT__( "\033[1;35m" ) );
468 }
469
470 response -> println( __CONST_TXT__( "\r\nBroken pipe!" ) );
471
472 if( formatting ){
473 response -> print( __CONST_TXT__( "\033[0;37m" ) );
474 }
475
476 response -> println( originalCommandData );
477
478 for( i = 0; i < brokenPipePos; i++ ){
479 response -> print( ' ' );
480 }
481
482 if( formatting ){
483 response -> print( __CONST_TXT__( "\033[1;31m" ) );
484 }
485
486 response -> println( "\u25B2" );
487
488 for( i = 0; i < brokenPipePos; i++ ){
489 response -> print( ' ' );
490 }
491
492 response -> print( "\u2514 The pipe broke here" );
493
494 if( formatting ){
495 response -> print( __CONST_TXT__( "\033[0;37m" ) );
496 }
497
498 }
499
500 return false;
501 }
502
503 // Check if piping is enabled.
504 // Check if we have at least one pipe left.
505 if( ( pipeArgBuffer != NULL ) && pipePos > 0 ){
506
507 // To remove whitespace from the new command begin.
508 while( tempBuff[ pipePos + 1 ] == ' ' ){
509 pipePos++;
510 }
511
512 // Execute the next command in the pipe.
513 if( executeCommand( &tempBuff[ pipePos + 1 ] ) ){
514 // If the command execution fails, we have to return to break recursion.
515 return false;
516 }
517
518 }
519
520 }
521
522 }
523
524 // If it is not an added function, we have to check for internal functions.
525 // 'help' is an internal function that prints the available commands in order.
526 else if( strcmp( tempBuff, (const char*)"help" ) == 0 ){
527
528 // We have to check for single or described help function.
529 if( strcmp( arg, (const char*)"-d" ) == 0 ){
530
531 printHelp( response, true, formatting );
532
533 }
534
535 else{
536
537 printHelp( response, false, formatting );
538
539 }
540
541 }
542
543 else{
544
545 // If we went through the whole tree and we did not found the command in it,
546 // we have to notice the user abut the problem. Maybe a Type-O
547 response -> print( __CONST_TXT__( "Command \'" ) );
548 response -> print( tempBuff );
549 response -> print( __CONST_TXT__( "\' not found!" ) );
550
551 }
552 return true;
553}

555bool Commander::execute( const char *cmd ){
556
557 // Default execute handler, so the default response will be chosen.
558 response = &defaultResponse;
559
560 // Reset the pipe position tracker.
561 pipeCounter = 0;
562
563 // Save the address of the original message.
564 originalCommandData = cmd;
565
566 // If piping is enabled, we have to flush the pipe channel before command execution.
567 if( pipeArgBuffer != NULL ){
568 while( pipeChannel -> available() ){
569 pipeChannel -> read();
570 }
571 }
572
573 // Execute the command.
574 return( executeCommand( cmd ) );
575

635Commander::API_t* Commander::operator [] ( const char* name ){
636
637 // Stores the next elements address in the tree
638 API_t *next;
639
640 // Stores the previous elements address in the tree
641 API_t *prev;
642
643 // It will store string compersation result
644 int comp_res;
645
646 prev = &API_tree[ 0 ];
647
648 comp_res = ( (*this).*(commander_strcmp_tree_ram) )( prev, name );
649
650 (comp_res > 0) ? (next = (prev->left)) : ( next = (prev->right));
651
652 // Go through the binary tree until you find a match, or until you find the
653 // end of the tree.
654 while( ( comp_res !=0 ) && ( next != NULL ) ){
655
656 prev = next;
657 comp_res = ( (*this).*(commander_strcmp_tree_ram) )( prev, name );
658 (comp_res > 0) ? (next = (prev->left)) : ( next = (prev->right));
659
660 }
661
662 // If comp_res variable has a zero in it, that means in the last iteration
663 // we had a match.
664 if( comp_res == 0 ){
665
666 return prev;
667
668 }
669
670 // If we did not found the command we return NULL.
671 return NULL;
672
673}

676
677 uint32_t i;
678
679 if( style ){
680 out -> println( __CONST_TXT__( "\033[1;31m----\033[1;32m Available commands \033[1;31m----\033[0;37m\r\n" ) );

709
710 out -> print( __CONST_TXT__( ": " ) );

712 if( memoryType == MEMORY_REGULAR ){

727int Commander::hasChar( const char* str, char c, bool ignoreString ){
728
729 int cntr = 0;
730
731 while( str[ cntr ] ){
732
733 // Check if we have to ignore characters inside a string.
734 // If so, we have to check if we are in a string.
735 if( ignoreString && inString( str, cntr ) ){
736 // In this case we can skip to the next character.
737 cntr++;
738 continue;
739 }
740
741 // Check if the current character is the searched one.
742 if( str[ cntr ] == c ){
743 // If it is, return with its position( index ).
744 return cntr;
745
746 }
747
748 // Increment the counter.
749 cntr++;
750
751 }
752
753 // If we are here, that means, we did not found the character in the string :(
754 // Return with an invalid address.
755 return -1;
756
757}

759int Commander::hasChar( const char* str, char c, int number, bool ignoreString ){
760
761 // Generic counter.
762 int i;
763
764 // It will store the current result.
765 int result;
766
767 // It will store te total position.
768 int cntr = 0;
769
770 // Try to find the number'th occurrence in the input string.
771 for( i = 0; i < number; i++ ){
772
773 result = hasChar( &str[ cntr ], c, ignoreString );
774
775 if( result < 0 ){
776 return -1;
777 }
778
779 cntr += ( result + 1 );
780
781 }
782
783 return cntr - 1;
784
785}

858void Commander::printSystemVariable( Stream* channel_p, const char* name, int decimalPlaces ){
859
860 SystemVariable_t* var;
861
862 if( channel_p == NULL ){
863 return;
864 }
865
866 var= getSystemVariable( name );
867
868 if( var == NULL ){
869
870 return;
871
872 }
873
874 if( ( var -> floatData ) != NULL ){
875 channel_p -> print( *var -> floatData, decimalPlaces );
876 return;
877 }
878
879 if( ( var -> intData ) != NULL ){
880 channel_p -> print( *var -> intData );
881 return;

967void Commander::update( char* buffer, int bufferSize, Stream* channel_p ){
968
969 // It will hold the nex incoming character from the channel.
970 char c;
971
972 if( bufferSize <=0 ){
973 return;
974 }
975
976 if( channel_p == NULL ){
977 return;
978 }
979
980 // Check if there is any data incoming.
981 while( channel_p -> available() ){
982
983 // Read the next incoming character.
984 c = channel_p -> read();
985
986 // Every command from Serial is terminated with a new-line
987 // character. If a new-line character arrives, we have to
988 // terminate the string in the commandFromSerial buffer,
989 // and execute it. After execution, we have to reset the
990 // commandIndex counter to zero.
991 if( c == '\r' ){
992 buffer[ updateBufferCounter ] = '\0';
993 channel_p -> println();
994 execute( buffer, channel_p );
995 updateBufferCounter = 0;
996 channel_p -> print( "\r\n$: " );
997 }
998
999 // If we have a carriage-return character we simply
1000 // ignore it.
1001 else if( c == '\n' ){
1002 continue;
1003 }
1004
1005 // Handle backspace events.
1006 else if( ( c == '\b' ) || ( c == 127 ) ){
1007 if( updateBufferCounter > 0 ){
1008 updateBufferCounter--;
1009 channel_p -> print( "\b \b" );
1010 }
1011 }
1012
1013 // Every other case we just put the data to the next
1014 // free space in the commandFromSerial buffer, increment
1015 // the commandIndex, and check if it wants to overflow.
1016 else{
1017 buffer[ updateBufferCounter ] = c;
1018 updateBufferCounter++;
1019 if( updateBufferCounter >= bufferSize ){
1020 updateBufferCounter = bufferSize - 1;
1021 }
1022 else{
1023 channel_p -> print( c );
1024 }
1025 }
1026
1027 }
